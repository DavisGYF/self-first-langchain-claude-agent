### 3. 💬 多模态融合
```ts
// 将识别的文字与用户输入融合
const handleTextExtracted = (text: string) => {
  if (!input.trim()) {
    setInput(text);
  } else {
    setInput(prev => prev + '\n\n从图片中识别的文字：\n' + text);
  }
};
```
